Ejercicio 5
Código fuente: 
    Algoritmo sin_titulo
        Definir cant_personas, contador Como Entero
        Definir gH, gM Como Entero
        Definir ingreso_genero Como Caracter
        
        Mostrar "Ingese cantidad de personas:"
        leer cant_personas
        
        contador <- 0
        Mientras contador < cant_personas Hacer
            Mostrar "Persona nro. " contador+1 " de " cant_personas
            Mostrar "Ingrese género (H / M):"
            Leer ingreso_genero
            
            Si ingreso_genero == "h" o ingreso_genero == "H" Entonces
                gH <- gH + 1
            FinSi
            
            Si ingreso_genero == "m" o ingreso_genero == "M" Entonces
                gM <- gM + 1
            FinSi
            
            contador <- contador + 1
        FinMientras
        
        Mostrar "Cant. Mujeres: " gM
        Mostrar "Cant. Hombres: " gH
        Mostrar "Porcentaje de mujeres: " (100/cant_personas*gM) " %"
        Mostrar "Porcentaje de hombres: " (100/cant_personas*gH) " %"
    FinAlgoritmo
Ejercicio 6
Código fuente con "Mientras": 
    Algoritmo ejercicio6
        Definir nota, promGral Como Real
        Definir cantEstudiantes Como Entero
        Definir cantAprob, cantReprob Como Entero
        
        Mostrar "Escriba la primer nota: (entre 1 y 10; 0: salir)"
        Leer nota
        
        Mientras nota <> 0 Hacer
            cantEstudiantes <- cantEstudiantes + 1
            promGral <- promGral + nota
            
            Si nota < 6 Entonces
                cantReprob <- cantReprob + 1
            Sino
                cantAprob <- cantAprob + 1
            FinSi
            
            Mostrar "Escriba otra nota, entre 1 y 10 (0: salir)"
            Leer nota
        FinMientras
        
        promGral <- promGral / cantEstudiantes
        
        Mostrar "Cant. Estud. Aprobados: " cantAprob
        Mostrar "Cant. Estud. Reprobados: " cantReprob
        Mostrar "Prom. Gral.:" promGral
    FinAlgoritmo
Código fuente con "Repetir": 
    Algoritmo ejercicio6
        Definir nota, promGral Como Real
        Definir cantEstudiantes Como Entero
        Definir cantAprob, cantReprob Como Entero
        
        Mostrar "Escriba la primer nota: (entre 1 y 10; 0: salir)"
        Leer nota
        
        Repetir
            cantEstudiantes <- cantEstudiantes + 1
            promGral <- promGral + nota
            
            Si nota < 6 Entonces
                cantReprob <- cantReprob + 1
            Sino
                cantAprob <- cantAprob + 1
            FinSi
            
            Mostrar "Escriba otra nota, entre 1 y 10 (0: salir)"
            Leer nota
        Hasta Que nota = 0
        
        promGral <- promGral / cantEstudiantes
        
        Mostrar "Cant. Estud. Aprobados: " cantAprob
        Mostrar "Cant. Estud. Reprobados: " cantReprob
        Mostrar "Prom. Gral.:" promGral
    FinAlgoritmo
